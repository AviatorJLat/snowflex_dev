**Section 1: DBConnection bridge callbacks** (write from scratch, specific to our types):

```elixir
defmodule SnowflexDev.Ecto.Adapter.Connection do
  @behaviour Ecto.Adapters.SQL.Connection

  alias SnowflexDev.Query

  # --- DBConnection Bridge ---

  @impl true
  def child_spec(opts) do
    DBConnection.child_spec(SnowflexDev.Connection, opts)
  end

  @impl true
  def prepare_execute(conn, name, statement, params, opts) do
    query = %Query{name: name, statement: statement}
    DBConnection.prepare_execute(conn, query, params, opts)
  end

  @impl true
  def query(conn, statement, params, opts) do
    query = %Query{statement: IO.iodata_to_binary(statement)}
    case DBConnection.execute(conn, query, params, opts) do
      {:ok, _query, result} -> {:ok, result}
      {:error, _} = error -> error
    end
  end

  @impl true
  def query_many(conn, statement, params, opts) do
    case query(conn, statement, params, opts) do
      {:ok, result} -> {:ok, [result]}
      {:error, _} = error -> error
    end
  end

  @impl true
  def execute(conn, query, params, opts) do
    DBConnection.execute(conn, query, params, opts)
  end

  @impl true
  def stream(conn, statement, params, opts) do
    query = %Query{statement: IO.iodata_to_binary(statement)}
    DBConnection.prepare_stream(conn, query, params, opts)
  end

  @impl true
  def explain_query(conn, query, params, opts) do
    explain_sql = ["EXPLAIN ", query]
    case query(conn, explain_sql, params, opts) do
      {:ok, result} -> {:ok, result}
      {:error, _} = error -> error
    end
  end

  @impl true
  def to_constraints(_error, _opts), do: []

  @impl true
  def table_exists_query(table) do
    {"SELECT 1 FROM information_schema.tables WHERE table_name = ? LIMIT 1", [table]}
  end

  @impl true
  def ddl_logs(_result), do: []

  @impl true
  def execute_ddl(_command) do
    raise "execute_ddl is not supported by SnowflexDev"
  end
end
```
